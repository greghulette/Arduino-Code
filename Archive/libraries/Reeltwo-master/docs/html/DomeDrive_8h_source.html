319 int shortestDistance(int origin, int target)
320 {
321 int result = 0.0;
322 int diff = fmod(fmod(abs(origin - target), 360), 360);
323
324 if (diff > 180)
325 {
326 //There is a shorter path in opposite direction
327 result = (360 - diff);
328 if (target > origin)
329 result *= -1;
330 }
331 else
332 {
333 result = diff;
334 if (origin > target)
335 result *= -1;
336 }
337 return result;
